/* criar o banco imobiliaria*/

create table prorpietario (
    cpf         int         	 not null,
    nome        varchar(50)     not null,
    endereco    varchar(50)     not null,
    cidade      varchar(50)     not null,
    estado      char(2)         not null,
    telefone    int        		 not null,
    primary key(cpf));

create table corretor (
    cpf         int          	 not null,
    nome        varchar(50)      not null,
    endereco    varchar(50)      not null,
    cidade      varchar(50)      not null,
    estado      char(2)          not null,
    telefone    int          	 not null,
    primary key(cpf));

create table inquilino(
    cpf         int             	 not null,
    nome        varchar(50)         not null,
    endereco    varchar(50)         not null,
    cidade      varchar(50)         not null,
    estado      char(2)             not null,
    telefone    int             	 not null, 
    cpfcorretor int             	 not null,
    primary key (cpf),
    foreign key (cpfcorretor) references corretor(cpf));

create table imovel (
    codigo          int         	 not null,
    descricao       varchar(50)     not null,
    endereco        varchar(50)     not null,
    cidade          varchar(50)     not null,
    estado          char(2)         not null,
    tipo            varchar(50)     not null,
    valor           decimal(10,2)     		not null,
    cpfinquilino    int         	not null,
    cpfprorpietario int         	not null,
    primary key(codigo),
    foreign key (cpfinquilino) references inquilino(cpf),
    foreign key (cpfprorpietario) references prorpietario(cpf));
